# Props

Basic "Spinner" component

property propType required default custom
color string false #282c34 -
size string false 50 -
speed string false 2 -

# Usage



 




 

 
 
 
 


<template>
  <div>
    <Spinner />
  </div>
</template>

<script>
import { Spinner } from 'vuespins'

export default {
  name: 'app',
  components: { Spinner }
}
</script>

<template>
  <div>
    <Spinner color="#fb503b" size="80" speed="7" />
  </div>
</template>

<script>
import { Spinner } from 'vuespins'

export default {
  name: 'app',
  components: { Spinner }
}
</script>
